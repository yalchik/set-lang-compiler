// демонстрация составных выражений, консольного ввода/вывода

// 1. выводит 2 множества и предлагает пользователю ввести их симметрическую разность
// 2. высчитывает симметрическую разность этих множеств как объединение их разностей
// 3. сверяет введённое множество с вычисленным
// 4. если пользователь ошибся, то выводит правильный ответ

// определение входной функции
def main() {
	// инициализация множеств констанами
	var set1 = {1, 2, 3};
	var set2 = {2, 3, 4};
	// вывод в консоль
	write: set1;
	write: set2;
	// создание пустого множества
	var sdiff_user = {};
	// чтение из консоли
	read: sdiff_user;
	// вычисление симметрической разности
	var sdiff_test = (set1 - set2) + (set2 - set1);
	// проверка ответа
	var equal = sdiff_user == sdiff_test;
	// если ответ неверен, то выводим вычисленное значение
	if (equal == 0) {
		write: sdiff_test;
	}
}
